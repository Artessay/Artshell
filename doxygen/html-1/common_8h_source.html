53template<typename T>
54int Binary_Search(int left, int right, T val, T array[], int cmp(T a, T b))
55{
56 while (left < right)
57 {
58 int mid = (left + right) >> 1;
59 int compare_result = cmp(val, array[mid]);
60 if (compare_result == 0)
61 return mid;
62 else if (compare_result > 0)
63 left = mid + 1;
64 else
65 right = mid;
66 }
67
68 return -1;
69}
