29int Display::InputCommand(char *input, const int len)
30{
31 tcsetpgrp(STDIN_FILENO, getpid());
32
33 // 初始化输入缓冲器与相关变量
34 char ch;
35 int i = 0;
36 memset(input, 0, len);
37
38 // 循环读入字符
39 do
40 {
41 ssize_t state = read(console_->input_file_descriptor, &ch, 1);
42 if (state == 0)
43 {
44 // 读到了EOF，结束
45 if (i == 0) // 如果此时缓冲器中什么内容也没有
46 return 0; // 直接返回
47 else // 这是文本未加入换行的最后一行
48 {
49 input[i++] = '\n'; // 手动加入换行
50 return i; // 将最后一行命令处理完毕
51 }
52 }
53 else if (state == -1)
54 {
55 throw "Read Input Error";
56 }
57
58
59 if (ch == '\\') // 如果读到换行输入\命令就跳过继续
60 {
61 ch = getchar(); // 将随后的换行符读入
62 continue;
63 }
64
65 if (ch == ';') // 将；视为换行符，便于lexer和parser处理
66 {
67 ch = '\n';
68 perform = false;
69 }
70
71 input[i++] = ch;
72
73 if (i == len) // 达到最大长度了
74 {
75 buffer_ = "\e[1;31mERROR\e[0m input compand exceeds maximum length. 输入命令的长度超过了允许的最大长度";
76 memset(input, 0, len); // 清空缓冲区输入
77 return -1;
78 }
79 } while (ch != '\n');
80
81 #ifdef _DEBUG_
82 printf("input: %s", input);
83 #endif
84
85 return i;
86}

88void Display::render()
89{
90 buffer_ = ""; // 每轮循环前将输出缓冲区清空
91
92 // 如果不是从标准输入中输入或是不是将内容输出到标准输出的话，
93 if (console_->input_file_descriptor != STDIN_FILENO ||
94 console_->output_file_descriptor != STDOUT_FILENO)
95 return; // 就不需要打印提示符
96
97 if (!perform)
98 {
99 perform = true;
100 return;
101 }
102
103 int sret = 0;
104 const size_t len = strlen(console_->home);
105 if (strlen(console_->current_working_dictionary) >= len)
106 {
107 size_t i = 0;
108 while (i < len)
109 {
110 if (console_->current_working_dictionary[i] != console_->home[i])
111 break;
112 ++i;
113 }
114 if (i == len)
115 sret = i;
116 }
117
118 char buffer[BUFFER_SIZE]; // 打印缓冲区
119 sret = sret
120 ? snprintf(buffer, BUFFER_SIZE, "\e[1;32m%s@%s\e[0m:\e[1;34m~%s\e[0m> ", \
121 console_->user_name, console_->host_name, console_->current_working_dictionary+sret)
122 : snprintf(buffer, BUFFER_SIZE, "\e[1;32m%s@%s\e[0m:\e[1;34m%s\e[0m> ", \
123 console_->user_name, console_->host_name, console_->current_working_dictionary);
124 if (sret == -1)
125 {
126 throw "Error when writing into output buffer";
127 }
128
129 ssize_t ret;
130 ret = write(console_->output_file_descriptor, buffer, strlen(buffer));
131 if (ret == -1)
132 {
133 throw "Error when writing from buffer";
134 }
135
136 return;
137}
