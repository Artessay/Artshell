103void ProcessManager::PrintJobListDone(int output_fd)
104{
105 job_unit *pre_job = nullptr;
106
107 for (auto job : jobs)
108 {
109 #ifdef _DEBUG_
110 printf("Id: %u pid: %d\n", job.id, job.pid);
111 #endif
112 if (pre_job != nullptr) // 内存回收
113 {
114 this->JobRemove(pre_job);
115 pre_job = nullptr;
116 }
117
118 /* waitpid 在WNOHANG参数下 如果子进程已经结束，则返回子进程的pid；
119 如果子进程还未结束，则返回0； 如果发生错误，则返回-1 */
120 int stat_loc, wait_pid = waitpid(job.pid, &stat_loc, WNOHANG);
121 #ifdef _DEBUG_
122 printf("id: %u pid: %d wait: %d stat: %d\n", job.id, job.pid, wait_pid, stat_loc);
123 #endif
124 if (wait_pid == job.pid) // 已经结束
125 {
126 job.state = Done;
127 job.PrintJob();
128 pre_job = &job;
129 }
130 else if (wait_pid < 0) // 发生错误
131 {
132 throw std::exception();
133 }
134 }
135
136 if (pre_job != nullptr) // 内存回收
137 this->JobRemove(pre_job);
138}

176int ProcessManager::ForeGround(unsigned int jobid)
177{
178 for (auto job : jobs)
179 {
180 if (job.id == jobid)
181 {
182 Console::child_process_id = job.pid;
183 setpgid(job.pid, getgid());
184
185 // 将前端设置为子进程
186 tcsetpgrp(STDIN_FILENO, job.pid);
187 tcsetpgrp(STDOUT_FILENO, job.pid);
188 tcsetpgrp(STDERR_FILENO, job.pid);
189 job.state = Running;
190
191 kill(job.pid, SIGCONT);
192 while(waitpid(Console::child_process_id, NULL, WNOHANG) == 0 && Console::child_process_id >= 0);
193 Console::child_process_id = -1;
194
195 JobRemove(&job);
196
197 return jobid;
198 }
199 }
200
201 return -1;
202}
