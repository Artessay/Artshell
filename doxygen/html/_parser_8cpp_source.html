27bool Parser::shell_pipe(Console *model, Display* view, Executor* controller, int& argc, char *argv[], char *env[])
28{
29 int count = 0;
30 char *args[MAX_ARGUMENT_NUMBER];
31
32 int input_fd = model->GetInputFD(); // 记录下原始输入
33 int output_fd = model->GetOutputFD(); // 记录下原始输出
34
35 int i = 0;
36 do
37 {
38 if (strcmp(argv[i], "|") != 0) // 不是管道符
39 {
40 args[count] = argv[i];
41 count++;
42 }
43 else
44 {
45 args[count] = NULL; // 命令结束
46
47 int channel[2];
48 // channel[0] : read
49 // channel[1] : write
50 if (pipe(channel) == -1)
51 throw "Pipe Error, 错误终止";
52
53 #ifdef _DEBUG_
54 printf("channel: read %d write %d\n", channel[0], channel[1]);
55 #endif
56
57 pid_t pid = fork(); // 分裂进程，fork返回的是子进程的pid
58 if (pid < 0)
59 {
60 /* 错误处理 */
61 throw "Fork Error, 错误终止";
62 }
63 else if (pid == 0)
64 {
65 /* 子进程 */
66 setenv("parent", getenv("shell"), 1); // 设置调用子进程的父进程
67 close(channel[0]); // 关闭读进程
68 int fd = channel[1];
69
70 model->SetOutputFD(fd);
71 model->SetOutputRedirect();
72
73 dup2(fd, STDOUT_FILENO); // 重定向标准输出至channel[1]
74
75 shell_execute(model, view, controller, count, args, env);
76
77 /* 在shell execute里包含了重定向后文件的关闭等操作，此处可以不必再次关闭 */
78
79 return EXIT;
80 }
81 else
82 {
83 /* 父进程 */
84 wait(NULL); // 为保持逻辑，需要等子进程输出之后再继续
85
86 close(channel[1]);
87 int fd = channel[0];
88
89 model->SetInputFD(fd);
90 model->SetInputRedirect();
91
92 dup2(fd, STDIN_FILENO); // 重定向标准输入至fd
93
94 count = 0;
95 }
96
97 }
98
99 ++i;
100 } while (i < argc);
101
102 #ifdef _DEBUG_
103 printf("Parent Process\n");
104 #endif
105 /* 最后一条命令也要执行 */
106 args[count] = NULL; // 命令结束
107 bool exit_state = shell_execute(model, view, controller, count, args, env);
108
109 #ifdef _DEBUG_
110 printf("pipe: Input %d Output %d Error %d\n", model->GetInputFD(), model->GetOutputFD(), model->GetErrorFD());
111 #endif
112
113 /* 无论重定向如何发生，最后将其还原为本来的状态
114 可能执行时已关闭文件，但未正确设置原始状态 */
115 if (model->GetInputFD() != input_fd) // 如果发生了输入重定向且未关闭
116 {
117 // dup2(model->GetSTDIN(), STDIN_FILENO);
118 model->SetInputFD(input_fd); // 恢复输入
119 // model->ResetInputRedirect(); // 恢复状态
120 }
121
122 if (model->GetOutputFD() != output_fd) // 如果发生了输出重定向
123 {
124 // dup2(model->GetSTDOUT(), STDOUT_FILENO);
125 model->SetOutputFD(output_fd); // 恢复输出
126 // model->ResetOutputRedirect(); // 恢复状态
127 }
128
129 return exit_state;
130}

133{
134 if (argc == 0)
135 return 0; // 无参，不需处理
136
137 for (int index = argc-1; index > 0; --index) // 从末尾开始往前扫描，第一个不必扫
138 {
139 std::string arg(argv[index]); // 使用string类处理
140
144 if (arg == "<" || arg == "0<")
145 {
146 if (index + 1 == argc) // 重定向符号是最后一个输入
147 {
148 throw "语法解析错误";
149 }
150
151 if (model->GetInputRedirect()) // 如果已经设置了重定向状态了
152 {
153 throw "多重重定向错误";
154 }
155
156 const char * input_file = argv[index + 1];
157 int fd = open(input_file, O_RDONLY);
158 if (fd < 0)
159 throw std::exception();
160
161 model->SetInputFD(fd);
162 model->SetInputRedirect();
163
164 dup2(fd, STDIN_FILENO); // 重定向标准输入至fd
165
166 for (int jump = index + 2; jump < argc; ++jump)
167 argv[jump-2] = argv[jump];
168 argc = argc - 2;
169 argv[argc] = NULL;
170 }
171
173 if (arg == ">" || arg == "1>")
174 {
175 if (index + 1 == argc) // 重定向符号是最后一个输入
176 {
177 throw "语法解析错误";
178 }
179

181 {
182 throw "多重重定向错误";
183 }

190 model->SetOutputFD(fd);

231 if (arg == ">>" || arg == "1>>")

233 /* 词法解析时应该识别<和>符号的闭包 */
234 #ifdef _DEBUG_

268bool Parser::shell_execute(Console *model, Display* view, Executor* controller, int& argc, char *argv[], char *env[])
269{
270 // Argument_Display(argc, argv);
271
272 int input_fd = model->GetInputFD(); // 记录下原始输入
273 int output_fd = model->GetOutputFD(); // 记录下原始输出
274 int error_fd = model->GetErrorFD(); // 记录下原始错误输出
275
276 // 执行命令
277 try
278 {
279 // Parser 语法分析
280 Parser::shell_parser(model, view, controller, argc, argv, env);
281
282 // 执行命令
283 sh_err_t err = controller->execute(argc, argv, env);
284
285 // 根据返回状态判断
286 if (err == SH_EXIT)
287 {
288 view->show(); // 将退出信息显示
289 return true;
290 }
291 else if (err != SH_SUCCESS)
292 {
293 throw err;
294 }
295
296 view->show(); // 显示输出信息
297 view->clear(); // 清空结果
298 }
299 catch(const std::exception& e)
300 {
301 fprintf(stderr, "\e[1;31m[ERROR]\e[0m %s: %s\n", strerror(errno), e.what());
302 }
303 catch(const sh_err_t e)
304 {
305 fprintf(stderr, "\e[1;31m[ERROR]\e[0m MyShell: %s\n", shell_error_message(e));
306 }
307 catch(const char * message)
308 {
309 fprintf(stderr, "\e[1;31m[ERROR]\e[0m %s: %s\n", strerror(errno), message);
310 }
311 catch(...)
312 {
313 fprintf(stderr, "\e[1;31m[ERROR]\e[0m %s\n", strerror(errno));
314 }
315
316 if (model->GetInputRedirect()) // 如果发生了输入重定向
317 {
318 int state_code = close(model->GetInputFD()); // 关闭文件
319 if (state_code != 0) // 关闭错误处理
320 throw std::exception();
321
322 dup2(model->GetSTDIN(), STDIN_FILENO);
323 model->SetInputFD(input_fd); // 恢复输入
324 model->ResetInputRedirect(); // 恢复状态
325 }
326
327 if (model->GetOutputRedirect()) // 如果发生了输出重定向
328 {
329 int state_code = close(model->GetOutputFD()); // 关闭文件
330 if (state_code != 0) // 关闭错误处理
331 throw std::exception();
332
333 dup2(model->GetSTDOUT(), STDOUT_FILENO);
334 model->SetOutputFD(output_fd); // 恢复输出
335 model->ResetOutputRedirect(); // 恢复状态
336 }
337
338 if (model->GetErrorRedirect()) // 如果发生了错误输出重定向
339 {
340 int state_code = close(model->GetErrorFD()); // 关闭文件
341 if (state_code != 0) // 关闭错误处理
342 throw std::exception();
343
344 dup2(model->GetSTDERR(), STDERR_FILENO);
345 model->SetErrorFD(error_fd); // 恢复错误输出
346 model->ResetErrorRedirect(); // 恢复状态
347 }
348
349 return false;
350}

353static const char * shell_error_message(sh_err_t err)
354{
355 switch (err)
