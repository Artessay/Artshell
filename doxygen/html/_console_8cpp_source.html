49void SignalHandler(int signal_)
50{
51 switch (signal_)
52 {
53 case SIGINT: // Ctrl C 交互注意信号
54 #ifdef _DEBUG_
55 printf("Ctrl + C\n");
56 #endif
57 if (write(STDOUT_FILENO, "\n", 1) < 0)
58 throw std::exception();
59 // 当kill的pid < 0时 取|pid|发给对应进程组。
60 // kill(-getpid(), SIGINT);
61
62 // 子进程的CTRL C重置了，由子进程处理中断
63 break;
64
65 case SIGTSTP: // Ctrl Z 键盘中断
66 #ifdef _DEBUG_
67 printf("Ctrl + Z\n");
68 #endif
69 if (write(STDOUT_FILENO, "\n", 1) < 0)
70 throw std::exception();
71
72 if (Console::child_process_id >= 0)
73 {
74 setpgid(Console::child_process_id, 0);
75 kill(-Console::child_process_id, SIGTSTP);
76
77 unsigned int jobid = cp->AddJob(Console::child_process_id, Stopped, cp->argc, (char **)cp->argv);
78
79 // 打印当前进程
80 char buffer[BUFFER_SIZE];
81 snprintf(buffer, BUFFER_SIZE-1, "[%u] %d\n", jobid, Console::child_process_id);
82 if (write(cp->output_std_fd, buffer, strlen(buffer)) == -1)
83 throw std::exception();
84
85 snprintf(buffer, BUFFER_SIZE-1, "[%u]%c\tStopped\t\t\t\t\t", jobid, ' ');
86 if (write(cp->output_std_fd, buffer, strlen(buffer)) == -1)
87 throw std::exception();
88
89 // 参数打印
90 if (cp->argc > 0)
91 {
92 // 确保行末无多余的空格
93 if (write(cp->output_std_fd, cp->argv[0], strlen(cp->argv[0])) == -1)
94 throw std::exception();
95 for (int i = 1; i < cp->argc; ++i)
96 {
97 if (write(cp->output_std_fd, " ", 1) == -1) // 打印空格
98 throw std::exception();
99
100 // 打印参数
101 if (write(cp->output_std_fd, cp->argv[i], strlen(cp->argv[i])) == -1)
102 throw std::exception();
103 }
104 }
105 if (write(cp->output_std_fd, "\n", 1) == -1)
106 throw std::exception();
107
108 Console::child_process_id = -1;
109 }
110 break;
111
112 case SIGCHLD: // 子进程结束
113 // 父进程收到子进程退出命令后，回收子进程
114 // waitpid(-1, NULL, WNOHANG);
115 break;
116
117 default:
118 break;
119 }

123int Console::init()
124{
125 try
126 {
127 // 获取用户名称
128 struct passwd *pw = getpwuid(getuid());
129 if (pw == nullptr)
130 {
131 throw "get user database entry error";
132 }
133 memset(user_name, 0, BUFFER_SIZE);
134 strncpy(user_name, pw->pw_name, BUFFER_SIZE-1);
135
136 // 获取主机名称
137 int ret;
138 ret = gethostname(host_name, BUFFER_SIZE-1);
139 if (ret != 0)
140 {
141 throw "Error when getting host name";
142 }
143
144 // 获取当前工作目录
145 char *result;
146 result = getcwd(current_working_dictionary, BUFFER_SIZE);
147 if (result == NULL)
148 {
149 throw "Error when getting current working dictionary";
150 }
151
152 // 获取主目录
153 memset(home, 0, BUFFER_SIZE);
154 strncpy(home, getenv("HOME"), BUFFER_SIZE-1);
155
156 // 设置shell环境变量
157 strncpy(shell_path_env, current_working_dictionary, BUFFER_SIZE);
158 strncat(shell_path_env, "/myshell", BUFFER_SIZE);
159 setenv("shell", shell_path_env, 1);
160
161 // 设置掩码
162 umask_ = umask(022); // 获取默认掩码
163 umask(umask_); // 改回原来掩码
164
165 // 设置默认文件描述符
166 input_file_descriptor = STDIN_FILENO;
167 output_file_descriptor = STDOUT_FILENO;
168 error_file_descriptor = STDERR_FILENO;
169
170 // 设置重定向状态
171 redirect_input = false;
172 redirect_output = false;
173 redirect_error = false;
174
175 // 备份STD IO
176 input_std_fd = dup(STDIN_FILENO);
177 output_std_fd = dup(STDOUT_FILENO);
178 error_std_fd = dup(STDERR_FILENO);
179
180 // 获取进程
181 process_id = getpid();
182 child_process_id = -1; // 暂无子进程
183
184 // signal(SIGINT, SignalHandler); // Ctrl + C
185 signal(SIGTSTP, SignalHandler); // Ctrl + Z
186 signal(SIGCHLD, SignalHandler); // 子进程结束时发送给父进程的信号
187
188 // 屏幕shell从后台调用tcsetpcgrp时收到的信号
189 signal(SIGTTIN, SIG_IGN); // 屏蔽SIGTTIN信号
190 signal(SIGTTOU, SIG_IGN); // 屏蔽SIGTTOU信号
191 }
192 catch(const std::exception& e)
193 {
194 std::cerr << e.what() << '\n';
195 return 1;
196 }
197
198 return 0;
199}
